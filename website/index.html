5-Minute Setup

Drop-in integration for
any agent framework

One line of code. Zero configuration. Automatic logging for every agent action.

agent.py 
import agentaudit

# Initialize with your API key
audit = agentaudit.AgentAudit(api_key="aa_live_...")

# Log any agent action
audit.log(
    action="prompt_submitted",
    prompt="Analyze Q3 data",
    response="Q3 revenue up 23%",
    metadata={"model": "gpt-4", "tokens": 150}
)

# Done. Compliance scan ran automatically.
# Alert fired if PII was detected.

import { AgentAudit } from '@agentaudit/sdk';

// Initialize with your API key
const audit = new AgentAudit({
    apiKey: 'aa_live_...'
});

// Log any agent action
await audit.log({
    action: 'prompt_submitted',
    prompt: 'Analyze Q3 data',
    response: 'Q3 revenue up 23%',
    metadata: { model: 'gpt-4', tokens: 150 }
});

// Done. Compliance scan ran automatically.
// Alert fired if PII was detected.

curl -X POST https://api.agentaudit.io/api/v1/audit-logs \
  -H "X-API-Key: aa_live_..." \
  -H "Content-Type: application/json" \
  -d '{
    "action": "prompt_submitted",
    "prompt": "Analyze Q3 data",
    "response": "Q3 revenue up 23%",
    "metadata": {"model": "gpt-4", "tokens": 150}
  }'
